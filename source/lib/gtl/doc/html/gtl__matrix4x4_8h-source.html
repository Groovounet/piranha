00353                 T ux = Axis (X);

00355                 T uz = Axis (Z);
00356     
00357                 Matrix.m_Value[0]  = c + (1 - c) * ux;
00358                 Matrix.m_Value[1]  = (1 - c) * ux * uy + s * uz;
00359                 Matrix.m_Value[2]  = (1 - c) * ux * uz - s * uy;

00362                 Matrix.m_Value[4]  = (1 - c) * uy * ux - s * uz;
00363                 Matrix.m_Value[5]  = c + (1 - c) * pow (uy, 2);

00368                 Matrix.m_Value[9]  = (1 - c) * uz * uz - s * ux;
